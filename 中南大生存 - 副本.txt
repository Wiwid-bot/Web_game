        function updateStatsUI() {
            document.getElementById('moneyVal').textContent = GameState.money;
            document.getElementById('knowledgeVal').textContent = GameState.knowledge;
            document.getElementById('energyVal').textContent = GameState.energy;
            document.getElementById('mentalityVal').textContent = GameState.mentality;
            document.getElementById('apVal').textContent = GameState.actionPoints;
            document.getElementById('dayVal').textContent = GameState.day;
            document.getElementById('locationTag').textContent = '📍 ' + (locations[GameState.currentLocation]?.name || '未知');
        }

        // 绘制场景
        function renderScene(locationId) {
            const sceneArt = document.getElementById('sceneArt');
            const sceneLabel = document.getElementById('sceneLabel');
            const loc = locations[locationId];
            sceneLabel.textContent = loc.name;
            sceneArt.innerHTML = '';
            // 根据地点生成像素场景元素
            switch(locationId) {
                case 'wuyang_square':
                    sceneArt.innerHTML = `<div class="pixel-sun" style="top:12px;right:30px;"></div>
                        <div class="pixel-building" style="bottom:20px;left:20px;width:90px;height:80px;background:#c0d8c8;"></div>
                        <div class="pixel-building" style="bottom:20px;right:30px;width:70px;height:100px;background:#b8d0bc;"></div>
                        <div style="position:absolute;bottom:20px;left:50%;transform:translateX(-50%);font-size:38px;">🏛️</div>`;
                    break;
                case 'library':
                    sceneArt.innerHTML = `<div class="pixel-building" style="bottom:20px;left:50%;transform:translateX(-50%);width:160px;height:110px;background:#d4e2d8;border:4px solid #5e8b6f;"></div>
                        <div style="position:absolute;top:30px;left:60px;font-size:24px;">📚</div>`;
                    break;
                case 'wenbo':
                    sceneArt.innerHTML = `<div class="pixel-building" style="bottom:20px;right:40px;width:70px;height:130px;background:#c4d4c0;"></div>
                        <div style="position:absolute;top:10px;right:50px;font-size:40px;">🕰️</div>`;
                    break;
                case 'canteen_xiaohu':
                    sceneArt.innerHTML = `<div style="position:absolute;bottom:20px;width:100%;text-align:center;font-size:48px;">🍜</div>`;
                    break;
                case 'nanhu_road':
                case 'xiaonan_lake':
                    sceneArt.innerHTML = `<div class="pixel-water" style="bottom:0;width:100%;height:80px;"></div>
                        <div style="position:absolute;top:20px;right:50px;font-size:36px;">🌳</div>`;
                    break;
                case 'wentai':
                    sceneArt.innerHTML = `<div class="pixel-building" style="bottom:20px;left:50px;width:100px;height:90px;background:#c8d8c0;"></div>`;
                    break;
                case 'dormitory':
                    sceneArt.innerHTML = `<div style="position:absolute;bottom:20px;width:100%;text-align:center;font-size:42px;">🛏️</div>`;
                    break;
                default: sceneArt.innerHTML = '<div style="font-size:48px;">🏫</div>';
            }
            // 随机文三水彩蛋（15%概率）
            if (Math.random() < 0.15 && !GameState.gameOver) {
                const ws = document.createElement('div');
                ws.className = 'wensanshui';
                ws.textContent = '🐾';
                ws.style.bottom = '30px';
                ws.style.left = rand(20, 80) + '%';
                ws.title = '点击文三水！';
                ws.addEventListener('click', (e) => {
                    e.stopPropagation();
                    triggerWensanshui();
                });
                sceneArt.appendChild(ws);
            }
        }

        function triggerWensanshui() {
            GameState.wensanshuiFound++;
            const gift = wensanshuiEvent.gift();
            if (GameState.backpack.length < GameState.maxBackpack) {
                GameState.backpack.push(gift);
            }
            applyEffect(wensanshuiEvent.effect);
            let msg = `🐾 文三水突然出现！它送给你【${gift.name}】。${gift.desc}。心态+8。`;
            if (GameState.wensanshuiFound >= 21) msg += '\n(你已集齐21个彩蛋！解锁隐藏成就)';
            setDialog(msg);
            updateStatsUI();
            renderScene(GameState.currentLocation);
        }

        function applyEffect(eff) {
            if (eff.money) GameState.money = clamp(GameState.money + eff.money, 0, 2000);
            if (eff.knowledge) GameState.knowledge = clamp(GameState.knowledge + eff.knowledge, 0, GameState.maxKnowledge);
            if (eff.energy) GameState.energy = clamp(GameState.energy + eff.energy, 0, GameState.maxEnergy);
            if (eff.mentality) GameState.mentality = clamp(GameState.mentality + eff.mentality, 0, GameState.maxMentality);
        }

        function setDialog(text) {
            document.getElementById('dialogText').innerHTML = text.replace(/\n/g, '<br>');
        }

        function clearChoices() {
            document.getElementById('choicesArea').innerHTML = '';
        }

        function addChoice(text, callback, hint = '') {
            const btn = document.createElement('button');
            btn.className = 'choice-btn';
            btn.innerHTML = text + (hint ? `<span class="btn-hint">${hint}</span>` : '');
            btn.addEventListener('click', callback);
            document.getElementById('choicesArea').appendChild(btn);
        }

        function disableAllChoices() {
            document.querySelectorAll('.choice-btn').forEach(b => b.classList.add('disabled'));
        }

        // 消耗行动点并检查结束日
        function spendAP(amount = 1) {
            GameState.actionPoints -= amount;
            updateStatsUI();
            if (GameState.actionPoints <= 0) {
                endDay();
            } else {
                showLocationActions();
            }
        }

        function endDay() {
            if (GameState.day >= GameState.maxDays) {
                triggerExam();
                return;
            }
            // 每日随机事件
            let eventMsg = '';
            if (Math.random() < 0.5) {
                const availableEvents = randomEvents.filter(e => !e.condition || e.condition());
                if (availableEvents.length > 0) {
                    const ev = availableEvents[Math.floor(Math.random() * availableEvents.length)];
                    applyEffect(ev.effect);
                    eventMsg = `\n【随机事件】${ev.title}：${ev.desc}`;
                }
            }
            GameState.day++;
            GameState.actionPoints = GameState.maxAP;
            updateStatsUI();
            setDialog(`🌙 第${GameState.day-1}天结束。你回到宿舍休息。${eventMsg}\n新的一天开始了，行动点已恢复。`);
            clearChoices();
            addChoice('继续', () => {
                setDialog('今天要去哪里呢？');
                showMapChoices();
            });
        }

        function triggerExam() {
            GameState.gameOver = true;
            const ending = getEnding();
            setDialog(`📝 考试日来临！\n你走进考场……\n\n最终成绩：<span style="color:${ending.color};font-size:18px;">${ending.grade}</span>\n${ending.title}\n${ending.desc}\n\n知识:${GameState.knowledge} 精力:${GameState.energy} 心态:${GameState.mentality} 生活费:${GameState.money}`);
            clearChoices();
            addChoice('🔄 重新开始', () => { resetGame(); startGame(); });
        }

        function showMapChoices() {
            clearChoices();
            setDialog('选择要去的地点：');
            for (const [id, loc] of Object.entries(locations)) {
                addChoice(`${loc.name}`, () => {
                    GameState.currentLocation = id;
                    updateStatsUI();
                    renderScene(id);
                    setDialog(`你来到了【${loc.name}】。${loc.desc}`);
                    showLocationActions();
                }, loc.desc.substring(0,15)+'...');
            }
        }

        function showLocationActions() {
            const loc = locations[GameState.currentLocation];
            clearChoices();
            setDialog(`你在【${loc.name}】。剩余行动点：${GameState.actionPoints}。选择行动：`);
            // 通用行动
            if (loc.actions.includes('study')) {
                const bonus = loc.studyBonus || 1.0;
                const k = Math.floor(rand(8, 16) * bonus);
                addChoice('📚 学习', () => {
                    applyEffect({ knowledge: k, energy: -rand(10,18) });
                    setDialog(`你在${loc.name}认真复习。知识+${k}，精力消耗。`);
                    spendAP(1);
                }, `知识+${k}`);
            }
            if (loc.actions.includes('eat')) {
                const cost = loc.mealCost || 8;
                const en = loc.mealEnergy || 18;
                addChoice('🍜 吃饭', () => {
                    if (GameState.money >= cost) {
                        applyEffect({ money: -cost, energy: en, mentality: 5 });
                        setDialog(`你在${loc.name}吃了一顿，花费${cost}元。精力+${en}，心态+5。`);
                    } else {
                        setDialog('生活费不足！试试拾荒或找便宜食堂。');
                        return;
                    }
                    spendAP(1);
                }, `花费${cost}元 精力+${en}`);
            }
            if (loc.actions.includes('rest')) {
                const rb = loc.restBonus || { energy: 10, mentality: 5 };
                addChoice('😴 休息', () => {
                    applyEffect(rb);
                    setDialog(`你在${loc.name}放松休息。精力+${rb.energy||0}，心态+${rb.mentality||0}。`);
                    spendAP(1);
                }, `精力+${rb.energy||0} 心态+${rb.mentality||0}`);
            }
            if (loc.actions.includes('social')) {
                addChoice('💬 社交', () => {
                    const m = rand(8,18);
                    applyEffect({ mentality: m, energy: -3 });
                    setDialog(`你和同学交流讨论，心情舒畅。心态+${m}。`);
                    spendAP(1);
                }, `心态提升`);
            }
            if (loc.actions.includes('exercise')) {
                const eff = loc.exerciseEffect || { energy: 8, mentality: 6 };
                addChoice('🏃 运动', () => {
                    applyEffect(eff);
                    setDialog(`你在南湖跑道挥洒汗水。精力+${eff.energy}，心态+${eff.mentality}。`);
                    spendAP(1);
                }, `精力+${eff.energy} 心态+${eff.mentality}`);
            }
            if (loc.actions.includes('explore')) {
    addChoice('🔍 探索拾荒', () => {
        const items = loc.exploreItems;
        if (items && items.length > 0) {
            const item = items[Math.floor(Math.random() * items.length)];
            if (GameState.backpack.length < GameState.maxBackpack) {
                GameState.backpack.push(item);
                applyEffect(item.effect || {});
                // 准备弹窗显示的内容（把风险也加进去）
                const popupItem = {
                    name: item.name,
                    desc: item.desc + (item.risk ? ' (注意：可能有' + item.risk + '风险)' : '')
                };
                showSearchPopup(popupItem);  // 用弹窗展示
            } else {
                setDialog('背包已满，无法拾取更多物品。');
            }
        } else {
            setDialog('这里没什么可捡的。');
        }
        spendAP(1);
    }, '随机获得物品（弹窗提示）');
}
            addChoice('🗺️ 返回地图', showMapChoices, '更换地点');
        }

        function toggleBackpack() {
            const panel = document.getElementById('backpackPanel');
            if (panel.style.display === 'block') {
                panel.style.display = 'none';
            } else {
                const content = document.getElementById('backpackContent');
                if (GameState.backpack.length === 0) {
                    content.innerHTML = '空空如也';
                } else {
                    content.innerHTML = GameState.backpack.map((item, idx) =>
                        `<div class="backpack-item">📦 ${item.name} - ${item.desc || ''}</div>`
                    ).join('');
                }
                panel.style.display = 'block';
            }
        }

        function resetGame() {
            GameState.day = 1;
            GameState.actionPoints = 4;
            GameState.money = 500;
            GameState.knowledge = 15;
            GameState.energy = 80;
            GameState.mentality = 70;
            GameState.currentLocation = 'wuyang_square';
            GameState.backpack = [];
            GameState.wensanshuiFound = 0;
            GameState.gameOver = false;
            GameState.endingTriggered = false;
            updateStatsUI();
            renderScene('wuyang_square');
            clearChoices();
            document.getElementById('backpackPanel').style.display = 'none';
        }

        function startGame() {
            document.getElementById('sceneArt').innerHTML = '<img src="images/wensanshui.png" style="width:100%;height:100%;object-fit:contain;display:block;position:relative;animation: bounce 1.5s infinite steps(6);">';
            document.getElementById('sceneLabel').textContent = '文三水';
            setDialog('🐾 文三水：欢迎来到中南财经政法大学！\n博文明理，厚德济世。\n在这个期末周，你不仅要复习考试，还得想办法活下去……\n生活费剩余500元，知识储备堪忧。\n在7天内，探索南湖校区，拾荒求生，冲刺期末！');
            clearChoices();
            addChoice('开始生存', () => {
                setDialog('第一天，你站在五羊广场。阳光正好。');
                showMapChoices();
            });
        }

        // 初始化
        window.addEventListener('load', () => {
            updateStatsUI();
            renderScene('wuyang_square');
            startGame();
        });

        // 暴露到全局
        window.toggleBackpack = toggleBackpack;
